Q 35.1
public class Cone{
	private double r;
	private double h;
	public final static double PI = 3.14159;
	private static int nbCones = 0;

Q 35.2
public Cone(double r, double h){
		this.r = r;
		this.h = h;
		this.nbCones += 1;
	}

	public Cone(){
		this(Math.random()*10,Math.random()*10);
	}

	public double getVolume(){
		return (1.0 / 3) * PI * r * r * h;
	}

	public String toString(){
		return "Cone r = " + r + " h = " + h + " V = " + getVolume();
	}

	public static int getnbCones(){
		return nbCones;
	} 

}

		System.out.println(c1);
		Cone c2 = new Cone();
		System.out.println(c2);
		System.out.println(Cone.getnbCones()); //Il s'affiche 2
	}
}

public class Trio{
	public int numero;
	public static int compteur = 0;
	public Personne[] trio;

	public Trio(){
		trio = new Personne[3];
		trio[0] = new Personne();
		trio[1] = new Personne();
		trio[2] = new Personne();
		compteur += 1;
		numero = compteur;

}
